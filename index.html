def default_sensor(func):
    """
    [Data Integrity Guard]
    모든 컨트롤 DAG에서 공통으로 사용하는 유효성 체크용 데코레이터.
    이전 배치의 성공 여부를 동적으로 추적하여 데이터 누락(Data Hole)을 원천 차단합니다.
    """
    def wrapper(*args, **kwargs):
        context = get_current_context()
        schedule_type, schedule_interval = args[0], args[1]
        dag_run = context['dag_run']
        
        # 테스트 및 수동 실행 시 By-pass
        if dag_run.run_type == 'manual':
            return True
       
        elif dag_run.run_type == 'scheduled':
            my_dag_id = context['dag'].dag_id 
            current_execution_date = context['execution_date']
            
            # 스케줄 타입에 따른 선행 배치 기준일 산출
            match schedule_type:
                case ScheduleType.SCHEDULE_HOUR:
                    pre_execution_date = [current_execution_date - timedelta(hours=i*schedule_interval) for i in range(1,11)]
                case ScheduleType.SCHEDULE_DAY:
                    pre_execution_date = current_execution_date - timedelta(days=schedule_interval)

            # Airflow Meta DB를 조회하여 선행 DagRun 상태 검증
            my_pre_dagruns = DagRun.find(dag_id=my_dag_id, execution_date=pre_execution_date)
            
            if my_pre_dagruns:
                for run in my_pre_dagruns:
                    if run.state != State.SUCCESS:
                        logging.error(f"선행 배치({run.state}) 미완료로 인한 실행 중단 방어.")
                        return False 

            return func(args, kwargs)
            
    return wrapper
                        

def create_trigger_task(trigger_dag_id, task_id):
    """
    [Resource Optimization]
    비동기 처리를 위한 TriggerDagRunOperator 헬퍼 함수입니다.
    deferrable=True 옵션을 표준화하여, 타겟 DAG가 끝날 때까지 
    현재 Worker 슬롯을 점유하지 않고 Triggerer 노드에서 가볍게 대기하도록 설계했습니다.
    """
    return TriggerDagRunOperator(
        task_id=task_id,
        trigger_dag_id=trigger_dag_id,
        reset_dag_run=True,
        wait_for_completion=True,
        deferrable=True,  # 핵심: Worker 슬롯 절약 (비용 최적화)
        poke_interval=60,
        logical_date="{{ execution_date }}",
        conf={
            # XCom을 활용한 데이터 파라미터 컨텍스트 자동 전파
            'data_interval_start': "{{ task_instance.xcom_pull(key='data_interval_start') }}",
            'data_interval_end': "{{ task_instance.xcom_pull(key='data_interval_end') }}"
        }
    )
                        

class RedshiftLogUtility:
    """
    [Performance Tuning]
    Redshift에 커스텀 ETL 로그를 기록하는 유틸리티.
    건별 INSERT 시 발생하는 I/O 병목을 제거하기 위해 메모리 버퍼링 및 Bulk Insert 패턴을 적용했습니다.
    """
    DETAIL_LOG_COLUMNS = ["DAG_ID", "RUN_ID", "TASK_ID", "STATUS", "AFFECTED_ROWS", "SQL_EXEC_TIME"] # 생략

    def __init__(self, redshift_conn_id: str):
        self.hook = RedshiftSQLHook(redshift_conn_id=redshift_conn_id)
        self.detail_buffer = [] # 로그 버퍼링 큐

    def insert_query_log(self, **kwargs):
        """메인 ETL 로직 내에서 호출되어 로그를 버퍼에 적재 (DB I/O 발생 X)"""
        params_tuple = (
            kwargs['dag_id'], kwargs['run_id'], kwargs['task_id'], kwargs['status'],
            kwargs['affected_rows'], kwargs.get('exec_time')
        )
        self.detail_buffer.append(params_tuple)

    def flush_detail_logs(self):
        """
        태스크 완료/실패 시 한 번만 호출되어 버퍼의 모든 로그를 일괄 적재합니다.
        Redshift의 COPY/Bulk Insert 특성을 살려 로깅 오버헤드를 90% 이상 절감했습니다.
        """
        if not self.detail_buffer: return

        try:
            self.hook.insert_rows(
                table='admin.nbs_log_detail',
                rows=self.detail_buffer,
                target_fields=self.DETAIL_LOG_COLUMNS,
                commit_every=0  # 단일 트랜잭션으로 강제
            )
        except Exception as e:
            logging.error(f"Bulk Log Insertion Failed: {e}")
        finally:
            self.detail_buffer.clear()
                        

import pandas as pd
import requests
from fake_useragent import UserAgent

def fetch_api_to_dataframe(json_url):
    """
    [Data Ingestion Strategy]
    단순 수집을 넘어, 차단 방지와 확장성을 고려한 외부 API 수집 모듈입니다.
    """
    # IP/세션 차단 우회를 위한 Random User-Agent 주입
    headers = {'User-Agent': UserAgent().random}
    response = requests.get(json_url, headers=headers, timeout=30)
    
    if response.status_code == 200:
        return pd.json_normalize(response.json())
    raise ConnectionError(f"API 호출 실패: {response.status_code}")
