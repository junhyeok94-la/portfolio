def default_sensor(func):
    """
    모든 컨트롤 DAG에서 공통으로 사용하는 유효성 체크용 데코레이터입니다.
    이전 배치가 성공적으로 완료되었는지 확인하여 데이터 정합성을 보장합니다.
    """
    def wrapper(*args, **kwargs):
        context = get_current_context()
        schedule_type = args[0]
        schedule_interval = args[1]
        
        dag_run = context['dag_run']
        
        # Manual 실행 시 유효성 체크 건너뜀 (테스트 편의성)
        if dag_run.run_type == 'manual':
            return True
       
        # Scheduled 실행 시 선행 DAG 상태 체크
        elif dag_run.run_type == 'scheduled':
            my_dag_id = context['dag'].dag_id 
            current_execution_date = context['execution_date']
            
            # 스케줄 타입에 따라 이전 실행 날짜 계산 (Hourly는 이전 10개까지 체크)
            match schedule_type:
                case ScheduleType.SCHEDULE_HOUR:
                    pre_execution_date = [current_execution_date - timedelta(hours=i*schedule_interval) for i in range(1,11)]
                case ScheduleType.SCHEDULE_DAY:
                    pre_execution_date = current_execution_date - timedelta(days=schedule_interval)
                # ... (Other cases omitted for brevity)

            # 이전 DagRun 조회 및 상태 검증
            my_pre_dagruns = DagRun.find(dag_id=my_dag_id, execution_date=pre_execution_date)
            
            if my_pre_dagruns:
                for run in my_pre_dagruns:
                    if run.state != State.SUCCESS:
                        logging.error(f"Pre-requisite DAG run is not success: {run.state}")
                        return False # Skip or Fail current run

            return func(args, kwargs)
            
    return wrapper
                        

def create_trigger_task(trigger_dag_id, task_id, env_type=None):
    """
    컨트롤 DAG에서 하위 서브 DAG를 트리거하기 위한 TriggerDagRunOperator 생성 함수.
    상위 DAG의 시간 변수(data_interval)를 하위 DAG로 안전하게 전달합니다.
    """
    return TriggerDagRunOperator(
        task_id=task_id,
        trigger_dag_id=trigger_dag_id,
        reset_dag_run=True,        # 재실행 시 DAG Run 초기화
        wait_for_completion=True,  # 하위 DAG 완료까지 대기
        deferrable=True,           # 워커 리소스 점유 없이 Triggerer에서 대기 (리소스 최적화 핵심)
        poke_interval=60,
        logical_date="{{ execution_date }}",
        conf={
            # 상위 DAG의 Time Interval 변수를 하위 DAG로 전달
            'data_interval_start': "{{ task_instance.xcom_pull(task_ids='set_time_interval', key='data_interval_start') }}",
            'data_interval_end': "{{ task_instance.xcom_pull(task_ids='set_time_interval', key='data_interval_end') }}",
            'exec_type': "{{ task_instance.xcom_pull(task_ids='set_time_interval', key='exec_type') }}"
        }
    )
                        

def etl_processor(conn_id, **context):
    """
    표준화된 DML SQL 실행 및 상세 로깅 처리 함수.
    별도의 보일러플레이트 코드 없이 SQL 파일만 있으면 실행 및 로깅이 자동화됩니다.
    """
    from hbu.include.nbs_log_handler import RedshiftLogUtility
    
    # ... (Context setup omitted)

    try:
        # 1. SQL 템플릿 렌더링 (Jinja Templating 적용)
        env = ti.task.get_template_env() 
        dml_content = env.get_template(f"sql/{task_id}.sql").render(
            **context, 
            start=time_param['data_interval_start'], 
            end=time_param['data_interval_end']
        )
        
        # 2. 로깅 유틸리티 초기화
        log_utility = RedshiftLogUtility(redshift_conn_id="hbu_lakehouse") 
        log_utility.insert_job_start_log(...)

        # 3. SQL 문별 순차 실행 및 개별 로깅
        for i, stmt in enumerate(statements, 1):
            try:
                dml_cursor.execute(stmt)      
                affected_rows = max(0, dml_cursor.rowcount)
                
                # 성공 로그 기록
                log_utility.insert_query_log(..., status="SUCCESS", affected_rows=affected_rows)
                
            except Exception as query_e:
                # 실패 로그 기록 및 트랜잭션 롤백
                log_utility.insert_query_log(..., status="FAILED", error_msg=str(query_e))
                dml_conn.rollback()
                raise AirflowFailException(f"Query Execution Failed: {query_e}")
        
        # 4. 최종 커밋 및 Job 종료 로그
        dml_conn.commit()
        log_utility.update_job_end_log(..., status='SUCCESS')

    except Exception as e:
        raise AirflowFailException(f"{e}")
    finally:
        if dml_conn: dml_conn.close()
                        

def attach_svltask_to_dbt_tasks(dag, dbt_tg, svl_models, dynamic_param):
    """
    Redshift Serverless를 사용하여 dbt run/test를 실행하는 Task를 동적으로 생성하고,
    기존 dbt TaskGroup과의 의존성을 자동으로 연결합니다.
    """
    # 환경별 Target 설정 (Dev/Prd)
    env = Variable.get("ENV")
    target_name = "prd_serverless" if env == "prd" else "dev_serverless"

    with dbt_tg:
        for model in svl_models:
            # Serverless 실행을 위한 BashCommand 생성 (표준화된 Arguments)
            bash_cmd = _get_dbt_bash_command(
                target_model=model["target_task"],
                target_name=target_name,
                ...
            )
            
            # dbt Run Task 생성
            target_task = BashOperator(
                task_id="run",
                bash_command=bash_cmd,
                doc_md=f"### Serverless Run: {model['target_task']}"
            )
            
            # dbt Test Task 생성 (옵션)
            if model.get('test', False):
                target_test_task = BashOperator(...)
                target_task >> target_test_task

                # DbtTaskGroup 내에서 선행 태스크를 찾아 연결

Solution: Deferrable Operator
- Worker 점유 시간 최소화
- Retry Exponential Backoff 적용

결과: 업데이트 기간 중 데이터 누락률 0%

Action: DB Cleanse Strategy
